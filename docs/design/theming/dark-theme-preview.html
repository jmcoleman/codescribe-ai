// Authentication Service
import { hash, compare } from 'bcrypt';
import { sign, verify } from 'jsonwebtoken';

export class AuthService {
  constructor(config) {
    this.secret = config.jwtSecret;
    this.saltRounds = 10;
  }

  async hashPassword(password) {
    return await hash(password, this.saltRounds);
  }

  async verifyPassword(password, hash) {
    return await compare(password, hash);
  }

  generateToken(userId) {
    return sign(
      { userId: userId },
      this.secret,
      { expiresIn: '7d' }
    );
  }
}

Authentication Service

A robust authentication service providing secure password hashing and JWT token generation for user authentication flows.

Installation

npm install bcrypt jsonwebtoken

Usage

import { AuthService } from './auth-service';

const auth = new AuthService({
  jwtSecret: 'your-secret-key'
});

// Hash a password
const hashed = await auth.hashPassword('user-password');
